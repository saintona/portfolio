<script>


/* GRID */

const grid = document.getElementById("grid");

const SIZE = 20;

const DENSITY = 0.08;

let dots = [];


function buildGrid(){

  grid.innerHTML="";

  dots=[];


  const cols = Math.ceil(window.innerWidth / SIZE);

  const rows = Math.ceil(window.innerHeight / SIZE);


  grid.style.gridTemplateColumns = `repeat(${cols}, ${SIZE}px)`;

  grid.style.gridTemplateRows = `repeat(${rows}, ${SIZE}px)`;


  for(let i=0;i<cols*rows;i++){

    const tile=document.createElement("div");

    tile.className="tile";


    const dot=document.createElement("div");

    dot.className="dot";


    tile.appendChild(dot);

    grid.appendChild(tile);


    dots.push(dot);

  }

}


function updateGrid(){

  dots.forEach(d=>{

    d.style.opacity = Math.random() < DENSITY ? 0.5 : 0;

  });

}


setInterval(updateGrid,2000);


/* ENTER */

const enterBtn = document.getElementById("enterBtn");

const enterScreen = document.getElementById("enterScreen");

const menu = document.getElementById("menu");

const subtext = document.getElementById("subtext");


enterBtn.onclick = () => {

  enterScreen.style.opacity = "0";

  enterScreen.style.transform = "scale(1.05)";


  setTimeout(() => {

    enterScreen.style.display = "none";

    menu.classList.add("active");

    subtext.style.opacity = "0.5";

  }, 600);

};


/* DATA */

const artImages = [

  "1art.jpg","2art.jpg","3art.jpg","4art.jpg","5art.jpg",

  "6art.jpg","7art.jpg","8art.jpg","9art.jpg","10art.jpg"

];


const fashionImages = [

  "1fashion.jpg","2fashion.jpg","3fashion.jpg","4fashion.jpg","5fashion.jpg",

  "6fashion.jpg","7fashion.jpg","8fashion.jpg","9fashion.jpg","10fashion.jpg"

];


const musicTracks = [

  "1music.mp3","2music.mp3","3music.mp3","4music.mp3","5music.mp3",

  "6music.mp3","7music.mp3","8music.mp3","9music.mp3","10music.mp3"

];


/* ELEMENTS */

const artBtn = document.getElementById("artBtn");

const fashionBtn = document.getElementById("fashionBtn");

const musicBtn = document.getElementById("musicBtn");


const modal = document.getElementById("modal");

const gallery = document.getElementById("gallery");


const viewer = document.getElementById("viewer");

const viewerImg = document.getElementById("viewerImg");


/* IMAGE GALLERY */

function openGallery(images){


  gallery.innerHTML="";


  images.forEach(src=>{

    const img=document.createElement("img");

    img.src=src;

    img.className="art-img";


    img.onclick=()=>{

      viewerImg.src=src;

      viewer.classList.add("active");

    };


    gallery.appendChild(img);

  });


  modal.classList.add("active");

}


/* MUSIC GALLERY */

function openMusic(tracks){


  gallery.innerHTML="";


  tracks.forEach(src=>{

    const wrap=document.createElement("div");

    wrap.className="audio-item";


    const audio=document.createElement("audio");

    audio.controls=true;

    audio.src=src;


    wrap.appendChild(audio);

    gallery.appendChild(wrap);

  });


  modal.classList.add("active");

}


/* BUTTONS */

artBtn.onclick = () => openGallery(artImages);

fashionBtn.onclick = () => openGallery(fashionImages);

musicBtn.onclick = () => openMusic(musicTracks);


/* CLOSE MODAL */

modal.onclick = (e)=>{

  if(e.target === modal){

    modal.classList.remove("active");

  }

};


/* CLOSE VIEWER */

viewer.onclick = ()=>{

  viewer.classList.remove("active");

};


buildGrid();

updateGrid();

window.addEventListener("resize", buildGrid);
